-- Auto Craft Shark Anchor mỗi 10 giây
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local CommF_ = ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("CommF_")

_G.AutoCraftSharkAnchor = true

task.spawn(function()
    while _G.AutoCraftSharkAnchor do
        pcall(function()
            CommF_:InvokeServer("Craft", "SharkAnchor", 1, {})
        end)
        
        task.wait(10)
    end
end)
