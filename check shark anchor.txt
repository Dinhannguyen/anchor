-- Script Check & Write File Soul Guitar by Cải
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local CommF_ = ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("CommF_")

-- Hàm kiểm tra xem đã có Soul Guitar chưa
local function hasSoulGuitar()
    local success, inventory = pcall(function()
        return CommF_:InvokeServer("getInventory")
    end)

    if success and inventory then
        for _, item in pairs(inventory) do
            -- Cải để tên chuẩn của game là Soul Guitar nhé
            if item.Name == "Soul Guitar" then
                return true
            end
        end
    end
    return false
end

-- Logic chính: Kiểm tra và xuất file
task.spawn(function()
    if hasSoulGuitar() then
        print("✅ Đã thấy Soul Guitar! Cải đang ghi file cho Bin...")
        
        -- Tên file và nội dung theo yêu cầu của Bin
        local fileName = "SoulGuitar_Status.txt"
        local content = "Completed-soul guitar"
        
        local writeSuccess, err = pcall(function()
            writefile(fileName, content)
        end)
        
        if writeSuccess then
            print("💾 File đã nằm gọn trong folder workspace rồi nhé Bin: " .. fileName)
        else
            warn("❌ Cải không ghi file được, lỗi: " .. tostring(err))
        end
    else
        print("❌ Bin chưa có Soul Guitar rồi, kiểm tra lại xem đã craft chưa nhé!")
    end
end)
