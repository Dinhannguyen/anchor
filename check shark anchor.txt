-- Script Loop Check & Write File Shark Anchor by Cải
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local CommF_ = ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("CommF_")

-- Biến kiểm soát vòng lặp
_G.MonitorAnchor = true 

print("Cải đã bắt đầu canh chừng Shark Anchor cho Bin...")

task.spawn(function()
    while _G.MonitorAnchor do
        -- 1. Quét Inventory
        local success, inventory = pcall(function()
            return CommF_:InvokeServer("getInventory")
        end)

        if success and inventory then
            local hasItem = false
            for _, item in pairs(inventory) do
                if item.Name == "Shark Anchor" then
                    hasItem = true
                    break
                end
            end

            -- 2. Nếu tìm thấy thì thực hiện các bước tiếp theo
            if hasItem then
                local username = LocalPlayer.Name
                local fileName = username .. ".txt"
                local content = "completed-anchor"
                
                local writeSuccess, err = pcall(function()
                    if writefile then
                        writefile(fileName, content)
                        return true
                    end
                end)

                if writeSuccess then
                    print("✅ Chúc mừng Bin! Đã có Shark Anchor và xuất file: " .. fileName)
                    -- 3. Đã xong việc thì Cải dừng vòng lặp luôn cho nhẹ máy
                    _G.MonitorAnchor = false 
                    break 
                end
            else
                print("🔍 Cải vẫn đang tìm... Bin cố lên, sắp có Mỏ neo rồi!")
            end
        end

        -- Đợi 15 giây rồi kiểm tra lại lần nữa
        task.wait(15)
    end
    print("🏁 Cải đã hoàn thành nhiệm vụ và nghỉ ngơi.")
end)
