local CommF_ = game.ReplicatedStorage.Remotes.CommF_
_G.AutoEquipGuitar = true

task.spawn(function()
    while _G.AutoEquipGuitar do
        local char = game.Players.LocalPlayer.Character
        
        if char and not char:FindFirstChild("Skull Guitar") then
            local inv = CommF_:InvokeServer("getInventory")
            
            for _,v in pairs(inv) do
                if v.Name == "Skull Guitar" then
                    CommF_:InvokeServer("LoadItem", "Skull Guitar")
                    break
                end
            end
        end
        
        task.wait(2)
    end
end)
